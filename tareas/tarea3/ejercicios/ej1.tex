\textbf{R:} Comenzemos por recordar que utilizaremos la función sigmoide y de las siguientes fórmulas:

\begin{align*}
    \hat{y} &= \sigma(z) = \frac{1}{1 + e^{-z}}\\
    % -
    z &= \beta_0 + \beta_1 x_1 + \cdots + \beta_n x_n \\
    J(\beta) &= -\frac{1}{n} \sum_{i=1}^n \left[ y_i \log(\hat{y}_i) + (1-y_i) \log(1-\hat{y}_i) \right]\\
    % -
    \frac{\partial J}{\partial \beta_j} &= 
    \frac{\partial J}{\partial \hat{y}_i} 
    \frac{\partial \hat{y}_i}{\partial z_i} 
    \frac{\partial z_i}{\partial \beta_j}  
    =\frac{1}{n} \sum_{i=1}^n (\hat{y}_i - y_i)x_{ij} \\
\end{align*}
Para la actualización tenemos
\begin{align*}
    \beta_0 &\gets \beta_0 - \eta \frac{\sum_{i=1}^n (\hat{y}_i - y_i)}{n} \\
    \beta_1 &\gets \beta_1 - \eta \frac{\sum_{i=1}^n (\hat{y}_i - y_i)x_{i1}}{n} \\
    \beta_2 &\gets \beta_2 - \eta \frac{\sum_{i=1}^n (\hat{y}_i - y_i)x_{i2}}{n}
\end{align*}
Donde $n$ es el número de muestras.\\
