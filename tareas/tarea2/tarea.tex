con eventos $X = x$, $Y = y$, entonces:
 
\[
p(Y = y \mid X = x) = \frac{p(X = x \mid Y = y)\,p(Y = y)}{\displaystyle\sum_{y_i \in Y} p(X = x \mid Y = y_i)\,p(Y = y_i)}
\]

\section{Ejercicio 9.}
Dado un modelo oculto de Markov $\mathrm{HMM} = \{\Sigma, S, A, B, \Pi\}$, sea
$w^{(1)}, \ldots, w^{(n)}$ una cadena de observaciones. Demostrar que:
 
\[
  p\!\left(w^{(1)} \cdots w^{(n)}\right)
  = \sum_{s^{(1)}} \cdots \sum_{s^{(T)}}
    \prod_{i=1}^{T}
    p\!\left(w^{(i)} \mid s^{(i)}\right)
    p\!\left(s^{(i)} \mid s^{(i-1)}\right)
\]
 
donde $s_1, \ldots, s_T \in S^{*}$ son cadenas de emisiones.

\section{Ejercicio 10.}
Demostrar que en el algoritmo de Viterbi las variables $\delta_i(t)$ cumplen la igualdad:
 
\[
  \delta_i(t)
  = \max_{y^{(1)} \cdots y^{(t-1)}}
    p\!\left(x^{(1)} \cdots x^{(t)},\; y^{(1)} \cdots y^{(t-1)},\; y^{(t)}\right)
\]
 
Y, por tanto, que para cualquier $n$ se tiene que el algoritmo regresa una cadena con probabilidad
máxima.
